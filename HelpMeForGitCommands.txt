0  git init //باید در مسیر پوشه این دستور را زد تا پوشه را به عنوان گیت محلی بشناسد 
   
1  git config --global user.name // یوزر فعلی را نشان میدهد ----اگر میخواهیم یوزر را معرفی کنیم بای در آخر بعد از یک فاصله یوزرمان را که در گیت ساخته ایم بدهیم
   
2  git config --global user.email 
  //مثل بالا برای ایمیل 
3  git clone git@github.com:mojmedia/GameRepo.git
 
	//در گیت هاب وقتی میخواهیم ب پروژه ای ملحق شویم در قسمت 
	//clon or download
	//آدرس اس اس اچ آنرا اینجا میگذاریم برای اینکه پروژخه را دانلود کند
4  git status //وضعیت فایلهای تغییر داده شده 
   
5  git add GameRepo/README.md // افزودن فایل مذکور به استیج 
6  cd GameRepo/
   // دستور سی دی داس
7  git fetch 
// آخرین تغییرات را میگیرد  
8  git pull  //آخرین فایلهای اضافه شده را میگیرد
9  git checkout developer_david  // در گیت هاب یا پوشه ی دولوپر_دیوید وجود دارد که میرود داخل آن وگرنه میسازد و میرود توش:) 
  
10  git add *
   // افزودن همه ی تغییرات به استیج

 //کامیت در ریپو محلی با پیغام
